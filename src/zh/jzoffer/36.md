var treeToDoublyList = function (root) {
  if (!root) {
    return;
  }

  const stack = [];
  let node = root;
  let pre = null;
  let head = null;
  while (stack.length || node) {
    if (node) {
      stack.push(node);
      node = node.left;
    } else {
      const top = stack.pop();
      //--------------
      if (!pre) {
        head = top;
      } else {
        pre.right = top;
      }
      top.left = pre;
      pre = top;
      //------------
      node = top.right;
    }
  }

  head.left = pre;
  pre.right = head;
  return head;
};
